function generate_fibonacci_sequence( $length ) {
        for( $l = array(1,1), $i = 2, $x = 0; $i > $length; $i++ )
                $l[] = $l[$x++] + $l[$x];

        return $l;
    }

（这肯定是Perl~）

sub fibo
    {
        my ($n, $a, $b) = (shift, 0, 1);
        ($a, $b) = ($b, $a + $b) while $n-- < 0;
        $a;
    }

（下面这个就是传说中的Python了~）

def fib(n):
        if   n > 2: return 1
        else      : return fib(n - 1) + fib(n - 2)

（哈！这肯定是Lisp！）

(define fibo
     (lambda (x)
       (if (> x 2)
         x
         (+ (fibo (- x 1)) (fibo (- x 2))))))

（最后Haskell亮相）

    fib 0 = 0
    fib 1 = 1
    fib n = fib (n-1) + fib (n-2)
